🔷 3. MARKET ENGINE (marketEngine.js)
let positions = [];
let pendingCodes = {};
let rollover = [];

let P_current = 1.0;
let totalVolume = 0;

let F = 0.5;
let H = 0.5;

// ======================
// CODE GENERATION
// ======================
function generateCode(userId, songId) {
  const code = `${songId}-${Math.random().toString(36).slice(2,6).toUpperCase()}`;

  pendingCodes[code] = {
    userId,
    songId,
    status: "pending",
    timestamp: Date.now()
  };

  return code;
}

// ======================
// CONFIRM CASH APP PAYMENT
// ======================
function confirmPayment(code, db) {
  const entry = pendingCodes[code];
  if (!entry) return { error: "Invalid code" };

  entry.status = "confirmed";

  positions.push({
    userId: entry.userId,
    songId: entry.songId,
    amount: 1,
    entryTime: Date.now(),
    isEarly: false,
    discount: 0
  });

  totalVolume += 1;

  db.credit(entry.userId, -1); // optional ledger sync

  return { success: true, entry };
}

// ======================
// ENTER DIRECTLY (internal use)
// ======================
function enterMarket(userId, songId) {
  positions.push({
    userId,
    songId,
    amount: 1,
    entryTime: Date.now(),
    isEarly: false
  });

  totalVolume++;
}

// ======================
// EARLY EXIT
// ======================
function earlyExit(userId, discount) {
  const pos = positions.find(p => p.userId === userId);
  if (!pos) return;

  pos.isEarly = true;
  pos.discount = discount;
}

// ======================
// FLOOR GOVERNOR
// ======================
function updateFloor(signal) {
  F = 0.5 + 0.4 * signal;
  H = 1 - F;
}

// ======================
// CLOSE MARKET (simple version)
// ======================
function closeMarket() {
  let floorPool = F * 1000;
  let remaining = floorPool;

  let payouts = [];

  let queue = [...positions];

  // early first
  queue.sort((a,b) => (b.discount || 0) - (a.discount || 0));

  for (let p of queue) {
    let payout = P_current * p.amount;

    if (remaining >= payout) {
      payouts.push({ userId: p.userId, payout });
      remaining -= payout;
    } else {
      payouts.push({ userId: p.userId, payout: remaining });
      remaining = 0;
      break;
    }
  }

  positions = [];
  totalVolume = 0;

  return payouts;
}

// ======================
function getState() {
  return {
    price: P_current,
    floor: F,
    house: H,
    volume: totalVolume,
    positions: positions.length,
    pendingCodes: Object.keys(pendingCodes).length
  };
}

module.exports = {
  generateCode,
  confirmPayment,
  enterMarket,
  earlyExit,
  updateFloor,
  closeMarket,
  getState
};

class MarketEngine {
  constructor() {
    this.P_current = 1.0;

    this.totalVolume = 0;
    this.targetVolume = 1000;

    this.demand = 0;
    this.supply = 0;

    this.queue = [];

    this.floorPercent = 0.5; // starts 50/50
    this.housePercent = 0.5;

    this.cycle = 1;
  }

  // ----------------------------
  // PRICE ENGINE (BREATHING SYSTEM)
  // ----------------------------
  updatePrice() {
    const imbalance = this.demand - this.supply;

    const ALPHA = 0.01; // sensitivity

    this.P_current += ALPHA * imbalance;

    if (this.P_current < 0.01) this.P_current = 0.01;

    return this.P_current;
  }

  // ----------------------------
  // ORDER ENTRY (BUY INTO SYSTEM)
  // ----------------------------
  enterMarket(userId, amount = 1) {
    this.queue.push({
      userId,
      amount,
      timestamp: Date.now(),
      status: "pending"
    });

    this.totalVolume += amount;
    this.demand += amount;

    return this.queue.length;
  }

  // ----------------------------
  // IMPULSE (LIQUIDITY SHOCK)
  // ----------------------------
  impulse(amount) {
    this.demand += amount;

    // strengthens floor pressure temporarily
    this.P_current += amount * 0.001;
  }

  // ----------------------------
  // SETTLEMENT ENGINE
  // ----------------------------
  settle() {
    if (this.totalVolume < this.targetVolume) return null;

    const settlementPrice = this.P_current;

    const floorPool = this.totalVolume * this.floorPercent;
    const housePool = this.totalVolume * this.housePercent;

    const result = {
      cycle: this.cycle,
      settlementPrice,
      roi: settlementPrice - 1,
      floorPool,
      housePool,
      queueSize: this.queue.length
    };

    this.resetCycle();

    return result;
  }

  // ----------------------------
  // RESET AFTER CLOSE
  // ----------------------------
  resetCycle() {
    this.P_current = 1.0;
    this.totalVolume = 0;
    this.demand = 0;
    this.supply = 0;
    this.queue = [];

    this.cycle += 1;
  }

  // ----------------------------
  // FLOOR/HOUSE GOVERNOR (YOUR RULE)
  // ----------------------------
  adjustBalance(floorPercent) {
    if (floorPercent < 0.5) floorPercent = 0.5;
    if (floorPercent > 0.9) floorPercent = 0.9;

    this.floorPercent = floorPercent;
    this.housePercent = 1 - floorPercent;
  }

  // ----------------------------
  // SAFE STOP (RISK CONTROL)
  // ----------------------------
  safeStop(threshold = 0.25) {
    if (this.P_current <= threshold) {
      return {
        stopped: true,
        price: this.P_current
      };
    }

    return { stopped: false };
  }
}

    });

  return { bids, asks };
}

function loadState(engines) {
  if (!fs.existsSync("state.json")) return;

  const data = JSON.parse(fs.readFileSync("state.json"));

  for (let id in data) {
    if (!engines[id]) continue;

    Object.assign(engines[id], data[id]);
  }
}

function buildMonitor(engines) {
  let totalVolume = 0;
  let totalPrice = 0;
  let activeMarkets = 0;

  let alerts = [];

  for (let id in engines) {
    const e = engines[id];

    totalVolume += e.totalVolume;
    totalPrice += e.P_current;

    if (e.totalVolume > 0) activeMarkets++;

    // ⚠️ alert conditions
    if (e.P_current < 0.05) {
      alerts.push(`LOW PRICE ALERT: Market ${id}`);
    }

    if (e.totalVolume > 900) {
      alerts.push(`NEAR SETTLEMENT: Market ${id}`);
    }
  }

  const avgPrice = totalPrice / Object.keys(engines).length;

  return {
    totalMarkets: Object.keys(engines).length,
    activeMarkets,
    totalVolume,
    avgPrice,
    alerts,
    time: Date.now()
  };
}

🔥 11A. SETTLEMENT FUNCTION

Add to engine.js:

settleMarket() {
  const payouts = [];

  const total = this.totalVolume;

  // floor percent (50% → 90%)
  const floor = Math.max(0.5, Math.min(this.floorPercent, 0.9));
  const floorPool = total * floor;

  let remainingPool = floorPool;

  while (this.queue.length > 0 && remainingPool > 0) {
    const user = this.queue.shift();

    const payout = Math.min(1.35, remainingPool); // capped payout logic

    payouts.push({
      userId: user.userId,
      payout,
      entry: user.amount,
      roi: payout - user.amount
    });

    remainingPool -= payout;
  }

  return payouts;
}

function renderQueue(queue) {
  const el = document.getElementById("queue");
  el.innerHTML = "";

  queue.forEach((q, i) => {
    const div = document.createElement("div");

    div.style.fontSize = "11px";
    div.style.padding = "2px";

    div.innerHTML = `
      #${i + 1} | ${q.userId} | $${q.amount}
    `;

    el.appendChild(div);
  });
}

const wallets = {};

function getWallet(userId) {
  if (!wallets[userId]) {
    wallets[userId] = {
      balance: 0,
      deposited: 0,
      positions: [],
      history: []
    };
  }

  return wallets[userId];
}

function requestWithdraw(userId, amount) {
  const w = getWallet(userId);

  if (w.balance < amount) {
    return { error: "NOT ENOUGH BALANCE" };
  }

  w.balance -= amount;

  w.history.push({
    type: "WITHDRAW_REQUEST",
    amount,
    time: Date.now()
  });

  return { ok: true };
}
