const axios = require('axios');
const fs = require('fs');
const path = require('path');

const API_KEY = process.env.KIE_AI_API_KEY; 
const SUBMIT_URL = 'https://api.kie.ai/v1/suno/submit';
const FETCH_URL = 'https://api.kie.ai/v1/suno/fetch/';

async function produceAndSaveBeat(prompt, fileName) {
  try {
    // 1. Submit Request
    const submitRes = await axios.post(SUBMIT_URL, {
      prompt: prompt,
      tags: "instrumental, studio-quality, beat",
      mv: "chirp-v3-5",
      make_instrumental: true
    }, { headers: { 'Authorization': `Bearer ${API_KEY}` } });

    const taskId = submitRes.data.data.task_id;
    console.log(`Production Started. Task ID: ${taskId}`);

    // 2. Polling loop
    let audioUrl = null;
    while (!audioUrl) {
      await new Promise(r => setTimeout(r, 10000)); // Wait 10s
      const check = await axios.get(`${FETCH_URL}${taskId}`, {
        headers: { 'Authorization': `Bearer ${API_KEY}` }
      });
      
      if (check.data.data.status === 'SUCCESS') {
        audioUrl = check.data.data.audio_url;
      } else if (check.data.data.status === 'FAILED') {
        throw new Error("Suno failed to generate.");
      }
      console.log("Checking status...");
    }

    // 3. Download the file to Replit
    console.log("Downloading beat to your production folder...");
    const writer = fs.createWriteStream(path.join(__dirname, fileName));
    const response = await axios({
        url: audioUrl,
        method: 'GET',
        responseType: 'stream'
    });

    response.data.pipe(writer);

    return new Promise((resolve, reject) => {
        writer.on('finish', () => {
            console.log(`SUCCESS: ${fileName} is now in your Replit files!`);
            resolve();
        });
        writer.on('error', reject);
    });

  } catch (err) {
    console.error("Error:", err.message);
  }
}

// RUN IT:
produceAndSaveBeat("Hard drill beat with dark piano", "new_fire_beat.mp3");
